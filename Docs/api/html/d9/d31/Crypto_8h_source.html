173std::vector<unsigned char> SharedSecretKeyGen(uint32_t node_id, EVP_PKEY* peerPublicKey) {
174 std::vector<unsigned char> sharedSecret;
175
176 // Load the node's private key
177 EVP_PKEY* privateKey = LoadPrivateKey(node_id);
178 if (!privateKey) {
179 std::cerr << "Failed to load private key for node " << node_id << std::endl;
180 return sharedSecret;
181 }
182
183 // Create derivation context
184 EVP_PKEY_CTX* ctx = EVP_PKEY_CTX_new(privateKey, nullptr);
185 if (!ctx) {
186 std::cerr << "Failed to create EVP_PKEY_CTX" << std::endl;
187 EVP_PKEY_free(privateKey);
188 return sharedSecret;
189 }
190
191 if (EVP_PKEY_derive_init(ctx) <= 0) {
192 std::cerr << "EVP_PKEY_derive_init failed" << std::endl;
193 EVP_PKEY_CTX_free(ctx);
194 EVP_PKEY_free(privateKey);
195 return sharedSecret;
196 }
197
198 if (EVP_PKEY_derive_set_peer(ctx, peerPublicKey) <= 0) {
199 std::cerr << "EVP_PKEY_derive_set_peer failed" << std::endl;
200 EVP_PKEY_CTX_free(ctx);
201 EVP_PKEY_free(privateKey);
202 return sharedSecret;
203 }
204
205 // Determine required buffer length
206 size_t secretLen = 0;
207 if (EVP_PKEY_derive(ctx, nullptr, &secretLen) <= 0) {
208 std::cerr << "EVP_PKEY_derive failed to get length" << std::endl;
209 EVP_PKEY_CTX_free(ctx);
210 EVP_PKEY_free(privateKey);
211 return sharedSecret;
212 }
213
214 sharedSecret.resize(secretLen);
215
216 // Derive shared secret
217 if (EVP_PKEY_derive(ctx, sharedSecret.data(), &secretLen) <= 0) {
218 std::cerr << "EVP_PKEY_derive failed" << std::endl;
219 sharedSecret.clear();
220 } else {
221 sharedSecret.resize(secretLen);
222 }
223
224 EVP_PKEY_CTX_free(ctx);
225 EVP_PKEY_free(privateKey);
226
227 return sharedSecret; // raw bytes
228}
